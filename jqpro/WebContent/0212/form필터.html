prop 는 상태를 나타내는거고 속성은 attribute 라고 함. 

		:enabled : 사용가능, 접근가능, 실행가능
		:disabled : 사용 불가능, 접근 불가능, 실행 불가능.
		
		$('form :enabled').css('border', '2px solid lime');
		$('form :disabled').css('border', '2px solid blue');
		  : form 뒤에 한칸 띄워야함.
		
		* 상태값 가져오기
		prop('상태속성이름') : true/false 리턴.
		
		* 상태값 변경
		prop('상태속성이름', true) : 해당 상태를 설정.
		prop('상태속성이름', false) : 해당 상태를 해제.
		
		prop('disabled', true) : 사용 불가능으로 변경.
		prop('disabled', false) : 사용 가능으로 변경 - enabled를 사용하지 않고 disabled false해야함.
