생성자 함수를 이용한 객체생성과 사용

		function Rect(width, height){
			this.width = width;
			this.height = height;
			
			this.getArea = function(){
				return this.width * this.height;
			}
		}
		
		var r1 = new Rect();
			객체의 속성 (width, height) 의 값이 정해지지않으면 undefinde가 발생한다.
			r1 객체의 속성에 값을 하나씩 대입한다.
			r1.width = 10;
			r1.height = 4;
				
		var r2 = new Rect(5,7);
		
		객체의 속성이나 메서드가  정해지지 않았어도 종적으로 추가해서 사용할 수 있다.
		r1.name = "사각형1";
		
		r2.name = "네모네모";
